# CIAO v0.2 完整工程设计文档

这份文档在原 PRD 基础上，补全了所有 coding agent 直接落地所需的技术细节：技术栈选型、目录结构、完整 API 契约、前端组件树、状态管理、实时通信协议、部署配置、以及逐文件实现规范。目标是让 agent 读完后无需再问"用什么框架"、"文件放哪里"、"接口长什么样"。

---

## 1. 技术栈与工具链

### 1.1 总览

| 层 | 选型 | 理由 |
|---|---|---|
| **前端框架** | Next.js 14 (App Router) | RSC + Server Actions，极简部署 |
| **UI 库** | Tailwind CSS + Radix UI | 留白、圆角、无边框卡片风格 |
| **状态管理** | Zustand (client) + React Server Components | 轻量、无 boilerplate |
| **实时通信** | Server-Sent Events (SSE) | 单向推送，比 WebSocket 轻 |
| **后端运行时** | Node.js 20 (Next.js API Routes) | 统一部署 |
| **数据库** | PostgreSQL 16 + Prisma ORM | 关系型，PRD 已定义 schema |
| **队列 / 后台任务** | BullMQ + Redis | Loop Engine 异步执行 |
| **AI Provider 抽象** | Vercel AI SDK + 自定义 adapter | 统一 streaming、tool use |
| **认证** | NextAuth.js v5 (Auth.js) | 简单、支持多 provider |
| **部署** | Vercel (前端) + Railway/Fly.io (worker) | 分离 web 与长时 worker |
| **Monorepo** | Turborepo | apps/web + packages/* |

## 3. 数据库 Schema（完整 Prisma）

```prisma
// apps/web/prisma/schema.prisma

generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "postgresql"
  url      = env("DATABASE_URL")
}

// ─── CAO & Workspace ───────────────────────────────────────

model Cao {
  id          String   @id @default(cuid())
  email       String   @unique
  name        String?
  avatarUrl   String?
  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt

  workspaces  WorkspaceMember[]
}

model Workspace {
  id          String   @id @default(cuid())
  name        String
  slug        String   @unique
  settings    Json     @default("{}")
  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt

  members     WorkspaceMember[]
  intents     Intent[]
  memories    Memory[]
  providers   ProviderConfig[]
}

model WorkspaceMember {
  id          String   @id @default(cuid())
  workspaceId String
  caoId       String
  role        String   @default("owner") // owner | admin

  workspace   Workspace @relation(fields: [workspaceId], references: [id], onDelete: Cascade)
  cao         Cao       @relation(fields: [caoId], references: [id], onDelete: Cascade)

  @@unique([workspaceId, caoId])
}

// ─── User-Facing Core ──────────────────────────────────────

model Intent {
  id              String   @id @default(cuid())
  workspaceId     String
  rawInput        String
  title           String
  interpretedGoal String
  mode            String   // ask | draft | act | ship | watch | review
  costMode        String   @default("balanced") // frugal | balanced | thorough
  state           String   @default("understanding")
  importance      String   @default("normal") // low | normal | high
  riskLevel       String   @default("unknown") // unknown | low | medium | high
  constraints     Json?    // string[]
  desiredOutcome  String?
  previewMessage  String?
  createdAt       DateTime @default(now())
  updatedAt       DateTime @updatedAt

  workspace       Workspace  @relation(fields: [workspaceId], references: [id], onDelete: Cascade)
  loops           Loop[]
  signals         Signal[]
  decisions       Decision[]
  outcomes        Outcome[]
  traces          TechnicalTrace[]

  @@index([workspaceId, state])
  @@index([workspaceId, createdAt])
}

model Signal {
  id          String   @id @default(cuid())
  intentId    String
  kind        String   // progress | risk | cost | confidence | blocker | decision | result
  level       String   // low | medium | high
  message     String
  compact     Boolean  @default(true)
  metadata    Json?
  createdAt   DateTime @default(now())

  intent      Intent @relation(fields: [intentId], references: [id], onDelete: Cascade)

  @@index([intentId, createdAt])
}

model Decision {
  id               String    @id @default(cuid())
  intentId         String
  title            String
  question         String
  recommendation   String?
  options          Json      // DecisionOption[]

  state            String    @default("open") // open | resolved | dismissed | expired
  resolvedOptionId String?
  resolvedAt       DateTime?
  expiresAt        DateTime?
  createdAt        DateTime  @default(now())

  intent           Intent @relation(fields: [intentId], references: [id], onDelete: Cascade)

  @@index([intentId, state])
}

model Outcome {
  id           String   @id @default(cuid())
  intentId     String
  title        String
  summary      String
  changed      Json?    // string[]
  verified     Json?    // string[]
  risks        Json?    // string[]
  confidence   String   // low | medium | high
  costSummary  Json?    // { mode, label, inputTokens?, outputTokens?, estimatedUsd? }
  receipt      Json?    // full receipt object
  state        String   @default("ready") // ready | accepted | reverted | archived
  createdAt    DateTime @default(now())

  intent       Intent @relation(fields: [intentId], references: [id], onDelete: Cascade)

  @@index([intentId])
}

model Memory {
  id            String    @id @default(cuid())
  workspaceId   String
  title         String
  trigger       String
  compactRule   String
  fullProcedure String?
  examples      Json?     // string[]

  status        String    @default("active") // active | disabled | archived

  lastUsedAt    DateTime?
  createdAt     DateTime  @default(now())
  updatedAt     DateTime  @updatedAt

  workspace     Workspace @relation(fields: [workspaceId], references: [id], onDelete: Cascade)

  @@index([workspaceId, status])
```

## 4. 完整 API 契约

### 4.1 CAO-facing API

#### `GET /api/home`

返回 Home 页面所需全部数据。

**Response:**

```ts
{
  greeting: string;           // "Good evening." 基于时间
  calmState: 'calm' | 'working' | 'needs_you' | 'attention';
  summary: string;            // "CIAO is calm." | "One decision needs you."
  now: NowCard[];             // max 3
  backgroundLoopCount: number; // 折叠的后台 loop 数
  decisions: DecisionCard[];  // open decisions, max 5
  outcomes: OutcomeCard[];    // recent ready outcomes, max 5
}

type NowCard = {
  intentId: string;
  title: string;
  state: string;
  message: string;
  costMode: string;
  risk: string;
  confidence: 'low' | 'medium' | 'high';
};

type DecisionCard = {
  id: string;
  intentId: string;
  intentTitle: string;
  title: string;
  question: string;
  recommendation: string | null;
  options: { id: string; label: string; description?: string; impact?: object }[];
  severity: string;
  createdAt: string;
};

type OutcomeCard = {
  id: string;
  intentId: string;
  title: string;
  summary: string;
  confidence: string;
  costLabel: string;
  state: string;
  createdAt: string;
};
```

---

#### `POST /api/intents`

创建新 Intent。

**Request:**

```ts
{
  rawInput: string;           // 必填，CAO 原始输入
  mode?: IntentMode;          // 可选，默认由 interpreter 推断
  costMode?: CostMode;       // 可选，默认 "balanced"
  importance?: 'low' | 'normal' | 'high';
  autoStart?: boolean;        // 默认 false，true 则跳过 preview 直接开始
}
```

**Response:**

```ts
{
  intent: Intent;
  preview: {
    title: string;
    interpretedGoal: string;
    mode: IntentMode;
    costMode: CostMode;
    constraints: string[];
    riskHints: string[];
    previewMessage: string;
  };
}
```

---

#### `GET /api/intents`

**Query params:** `?state=working,needs_decision&limit=20&offset=0`

**Response:** `{ intents: Intent[]; total: number }`

---

#### `GET /api/intents/:id`

**Response:**

```ts
{
  intent: Intent;
  currentSignals: Signal[];     // latest 5 signals
  activeLoopSummary: string | null;
  decisions: Decision[];        // open decisions for this intent
  outcomes: Outcome[];
}
```

---

#### `POST /api/intents/:id/pause`

**Request:** `{}` (no body needed)

**Response:** `{ intent: Intent; message: "Paused." }`

**Side effect:** Loop Engine pauses all running loops for this intent.

---

#### `POST /api/intents/:id/tighten`

**Request:** `{}` 

**Response:** `{ intent: Intent; message: "Tightened. Using more conservative approach." }`

**Side effect:** 
- costMode → 降一级 (thorough→balanced, balanced→frugal)
- Loop policy 增加保守性
- 减少上下文窗口

---

#### `POST /api/intents/:id/explore`

**Response:** `{ intent: Intent; message: "Exploring more options." }`

**Side effect:** 允许更多 search/plan loops，但不执行高风险动作。

---

#### `POST /api/intents/:id/deeper`

**Response:** `{ intent: Intent; message: "Going deeper with stronger analysis." }`

**Side effect:** 
- costMode → 升一级
- 允许 strong model
- 增加上下文预算

---

#### `POST /api/intents/:id/stop`

**Response:** `{ intent: Intent; outcome: Outcome | null; message: "Stopped. Summary generated." }`

**Side effect:** 停止所有 loops，生成当前发现的 Outcome。

---

#### `GET /api/decisions`

**Query:** `?state=open&limit=10`

**Response:** `{ decisions: DecisionCard[] }`

---

#### `POST /api/decisions/:id/resolve`

**Request:**

```ts
{
  optionId: string;  // 选择的 option id
}
```

**Response:** `{ decision: Decision; message: "Decision resolved. Loop resuming." }`

**Side effect:** Intent state 从 `needs_decision` → `working`，Loop Engine 恢复。

---

#### `GET /api/outcomes`

**Query:** `?state=ready&limit=20&offset=0`

**Response:** `{ outcomes: OutcomeCard[] }`

---

#### `POST /api/outcomes/:id/accept`

**Response:** `{ outcome: Outcome; message: "Accepted." }`

**Side effect:** Intent state → `accepted`。

---

#### `POST /api/outcomes/:id/revert`

**Response:** `{ outcome: Outcome; message: "Reverted." }`

#### `POST /api/outcomes/:id/save-memory`

#### `GET /api/events` (SSE)

Server-Sent Events 端点，推送实时更新。

**Event types:**

```ts
type SSEEvent =
```

```ts
// app/api/events/route.ts
import { NextRequest } from 'next/server';
import { getRedisSubscriber } from '@/lib/redis';

export async function GET(req: NextRequest) {
  const encoder = new TextEncoder();
  const workspaceId = req.headers.get('x-workspace-id'); // from auth

  const stream = new ReadableStream({
    async start(controller) {
      const sub = getRedisSubscriber();
      await sub.subscribe(`workspace:${workspaceId}:events`);

      sub.on('message', (channel, message) => {
        controller.enqueue(encoder.encode(`data: ${message}\n\n`));
      });

      // Heartbeat every 30s
      const heartbeat = setInterval(() => {
        controller.enqueue(encoder.encode(`: heartbeat\n\n`));
      }, 30000);

      req.signal.addEventListener('abort', () => {
        clearInterval(heartbeat);
        sub.unsubscribe();
        sub.disconnect();
        controller.close();
      });
    },
  });

  return new Response(stream, {
    headers: {
      'Content-Type': 'text/event-stream',
      'Cache-Control': 'no-cache',
      'Connection': 'keep-alive',
    },
  });
}
```

## 7. Loop Engine 详细设计

### 7.1 架构

Loop Engine 是 CIAO 的核心后端逻辑，运行在 BullMQ worker 中。

```
Intent created
  → BullMQ job: { type: 'process_intent', intentId }
  → Loop Worker picks up
  → Governor decides next step
  → Capability runs
  → Signal emitted
  → Repeat until Outcome or Decision needed
```

### 7.2 Governor 逻辑（伪代码）

```ts
// packages/engine/src/governor.ts

export async function determineNextStep(context: GovernorContext): Promise<NextStep> {
  const { intent, signals, loops, costMode, riskLevel, memories } = context;

  // Phase 1: Understanding
  if (intent.state === 'understanding') {
    if (!intent.interpretedGoal || intent.interpretedGoal === intent.rawInput) {
      return { type: 'run_capability', capability: 'interpret_intent' };
    }
    // Move to working
    await updateIntentState(intent.id, 'working');
  }

  // Phase 2: Risk gate check
  if (riskLevel === 'high' && !hasApprovedDecision(intent.id, 'risk_approval')) {
    return {
      type: 'ask_decision',
      decision: {
        title: 'High-risk area detected',
        question: `This touches ${getRiskDomain(signals)}. How should CIAO proceed?`,
        recommendation: 'review_only',
        options: [
          { id: 'review_only', label: 'Review only', impact: { risk: 'lower', speed: 'slower' } },
          { id: 'allow_draft', label: 'Allow draft patch', impact: { risk: 'normal', speed: 'normal' } },
          { id: 'full_access', label: 'Full access', impact: { risk: 'higher', speed: 'faster' } },
        ],
      },
    };
  }

  // Phase 3: Mode-specific logic
  switch (intent.mode) {
    case 'ask':
      return handleAskMode(context);
    case 'draft':
      return handleDraftMode(context);
    case 'act':
    case 'ship':
      return handleShipMode(context);
    case 'review':
      return handleReviewMode(context);
    case 'watch':
      return handleWatchMode(context);
  }
}

async function handleShipMode(ctx: GovernorContext): Promise<NextStep> {
  const { intent, loops } = ctx;
  const completedKinds = loops.filter(l => l.state === 'completed').map(l => l.kind);

  if (!completedKinds.includes('plan')) {
    return { type: 'run_capability', capability: 'plan_change' };
  }
  if (!completedKinds.includes('edit')) {
    return { type: 'run_capability', capability: 'edit_code' };
  }
  if (!completedKinds.includes('test')) {
    return { type: 'run_capability', capability: 'run_tests' };
  }
  if (!completedKinds.includes('review')) {
    return { type: 'run_capability', capability: 'review_diff' };
  }
  // All done
  return { type: 'build_outcome' };
}
```

```ts
export type ModelTier = 'small' | 'medium' | 'strong';

const MODEL_MAP: Record<string, Record<ModelTier, string>> = {
  openai: {
    small: 'gpt-4o-mini',
    medium: 'gpt-4o',
    strong: 'o1',
  },
  anthropic: {
    small: 'claude-3-haiku-20240307',
    medium: 'claude-sonnet-4-20250514',
    strong: 'claude-opus-4-20250514',
  },
};

export function selectModel(
  provider: string,
  preferredTier: ModelTier,
  capabilityKind: string,
  confidence: number
): { model: string; tier: ModelTier } {
  let tier = preferredTier;

  // Escalate if confidence is low and escalation is allowed
  if (confidence < 0.4 && tier !== 'strong') {
    tier = tier === 'small' ? 'medium' : 'strong';
  }

  // Simple capabilities can always use small
  const simpleCapabilities = ['summarize_result', 'extract_memory', 'monitor_signal'];
  if (simpleCapabilities.includes(capabilityKind) && tier !== 'strong') {
    tier = 'small';
  }

  return { model: MODEL_MAP[provider]?.[tier] ?? MODEL_MAP.openai[tier], tier };
}
```

### 7.5 Risk Sentinel

```ts
// packages/engine/src/risk-sentinel.ts

const HIGH_RISK_PATTERNS = [
  /auth/i, /authentication/i, /oauth/i,
  /billing/i, /payment/i, /charge/i,
  /permission/i, /rbac/i, /acl/i,
  /migration/i, /migrate/i,
  /secret/i, /credential/i, /api.?key/i,
  /production/i, /prod/i,
  /delete/i, /drop/i, /destroy/i, /truncate/i,
  /public.?api/i, /breaking.?change/i,
];

const MEDIUM_RISK_PATTERNS = [
  /database/i, /schema/i,
  /deploy/i, /release/i,
  /config/i, /environment/i,
  /user.?data/i, /pii/i,
];

export function assessRisk(intent: Intent, context?: string): RiskAssessment {
  const text = `${intent.rawInput} ${intent.interpretedGoal} ${context ?? ''}`;

  for (const pattern of HIGH_RISK_PATTERNS) {
    if (pattern.test(text)) {
      return { level: 'high', domain: pattern.source, requiresDecision: true };
    }
  }

  for (const pattern of MEDIUM_RISK_PATTERNS) {
    if (pattern.test(text)) {
      return { level: 'medium', domain: pattern.source, requiresDecision: false };
    }
  }

  return { level: 'low', domain: null, requiresDecision: false };
}
```

### 7.6 BullMQ Worker

```ts
// workers/loop-worker.ts

import { Worker, Queue } from 'bullmq';
import { determineNextStep } from '@ciao/engine/governor';
import { runCapability } from '@ciao/engine/capabilities';
import { buildOutcome } from '@ciao/engine/outcome-builder';
import { createDecision } from '@ciao/engine/decision-engine';
import { emitSignal } from '@ciao/engine/signal-engine';
import { prisma } from './prisma';
import { redis } from './redis';

const intentQueue = new Queue('intents', { connection: redis });

const worker = new Worker('intents', async (job) => {
  const { intentId } = job.data;
  const intent = await prisma.intent.findUnique({ where: { id: intentId } });
  if (!intent) return;

  // Governor loop - runs until outcome, decision, or pause
  let maxIterations = 30; // safety limit
  while (maxIterations-- > 0) {
    const context = await buildGovernorContext(intentId);
    const nextStep = await determineNextStep(context);

    switch (nextStep.type) {
      case 'run_capability': {
        const loop = await prisma.loop.create({
          data: {
            intentId,
            kind: mapCapabilityToLoopKind(nextStep.capability),
            state: 'running',
            costMode: intent.costMode,
            startedAt: new Date(),
          },
        });

        await emitSignal(intent.workspaceId, {
          type: 'loop_progress',
          data: { intentId, message: getProgressMessage(nextStep.capability) },
        });

        const result = await runCapability(nextStep.capability, {
          intentId,
          loopId: loop.id,
          costMode: intent.costMode,
        });

        await prisma.loop.update({
          where: { id: loop.id },
          data: { state: 'completed', completedAt: new Date() },
        });

        break;
      }

      case 'ask_decision': {
        const decision = await createDecision(intentId, nextStep.decision);
        await prisma.intent.update({
          where: { id: intentId },
          data: { state: 'needs_decision' },
        });
        await emitSignal(intent.workspaceId, {
          type: 'decision_created',
          data: decision,
        });
        return; // Exit worker, will resume when decision is resolved
      }

      case 'build_outcome': {
        const outcome = await buildOutcome(intentId);
        await prisma.intent.update({
          where: { id: intentId },
          data: { state: 'ready' },
        });
        await emitSignal(intent.workspaceId, {
          type: 'outcome_ready',
          data: outcome,
        });
        return;
      }

      case 'pause': {
        await prisma.intent.update({
          where: { id: intentId },
          data: { state: 'paused' },
        });
        await emitSignal(intent.workspaceId, {
          type: 'intent_state_changed',
          data: { intentId, state: 'paused' },
        });
        return;
      }
    }
  }
}, { connection: redis, concurrency: 5 });

// Resume handler - when a decision is resolved
export async function resumeIntent(intentId: string) {
  await prisma.intent.update({
    where: { id: intentId },
    data: { state: 'working' },
  });
  await intentQueue.add('process', { intentId });
}
```

```ts
  greeting: string;
  calmState: 'calm' | 'working' | 'needs_you' | 'attention';
  summary: string;
  now: NowCard[];
  backgroundLoopCount: number;
  decisions: DecisionCard[];
  outcomes: OutcomeCard[];
  isLoading: boolean;

  // Actions
  fetchHome: () => Promise<void>;
  handleEvent: (event: SSEEvent) => void;
};

export const useHomeStore = create<HomeState>((set, get) => ({
  greeting: '',
  calmState: 'calm',
  summary: '',
  now: [],
  backgroundLoopCount: 0,
  decisions: [],
  outcomes: [],
  isLoading: true,

  fetchHome: async () => {
    set({ isLoading: true });
    const res = await fetch('/api/home');
    const data = await res.json();
    set({ ...data, isLoading: false });
  },

  handleEvent: (event) => {
    switch (event.type) {
      case 'calm_state_changed':
        set({ calmState: event.data.calmState, summary: event.data.summary });
        break;
      case 'decision_created':
        set((s) => ({ decisions: [event.data, ...s.decisions].slice(0, 5) }));
        set({ calmState: 'needs_you' });
        break;
      case 'outcome_ready':
        set((s) => ({ outcomes: [event.data, ...s.outcomes].slice(0, 5) }));
        break;
      case 'intent_state_changed':
        // Refresh now cards
        get().fetchHome();
        break;
      case 'loop_progress':
        set((s) => ({
          now: s.now.map((n) =>
            n.intentId === event.data.intentId
              ? { ...n, message: event.data.message }
              : n
          ),
        }));
        break;
    }
  },
}));
```
